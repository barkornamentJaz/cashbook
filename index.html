const SS = SpreadsheetApp.getActiveSpreadsheet();

// ── Sheet name constants ──────────────────────────────
const SH_CB       = 'CashBook';
const SH_IOU      = 'IOU';
const SH_CC       = 'CashCount';
const SH_VOUCHERS = 'Vouchers';
const SH_META     = 'Meta';

// ── Column definitions (match spreadsheet headers) ───
// CashBook  A=Date B=Type C=Category D=Description E=Amount F=Reference G=Narration
const CB_FIELDS  = ['date','type','category','desc','amount','ref','narration'];
// IOU  A=Date B=IOU No C=Borrower D=Description E=Amount F=Expected Return G=Settled Date H=Status I=Notes
const IOU_FIELDS = ['date','no','borrower','desc','amount','returnDate','settledDate','status','notes'];
// Vouchers  A=Voucher No B=Date C=Payee D=Payment Method E=Category F=Description G=Amount(Rs) H=Cents I=Prepared By J=Approved By
const V_FIELDS   = ['no','date','payee','method','category','desc','rupees','cents','prepared','approved'];
// CashCount denominations → B2:B12
const CC_DENOMS  = [5000,2000,1000,500,100,50,20,10,5,2,1];

function doPost(e) {
  try {
    const d = JSON.parse(e.postData.contents);
    if (d.action === 'saveAll') {
      writeSheet(SH_CB,       d.cb,       CB_FIELDS);
      writeSheet(SH_IOU,      d.iou,      IOU_FIELDS);
      writeSheet(SH_VOUCHERS, d.vouchers, V_FIELDS);
      if (d.cashCounts) writeCashCount(d.cashCounts);
      writeMeta(d.meta);
    }
    return respond({saved: true});
  } catch(err) {
    return respond({saved: false, error: err.message});
  }
}

function doGet(e) {
  try {
    // Read opening balance directly from CashBook!I2
    const cbSh = SS.getSheetByName(SH_CB);
    const cbOpen = cbSh ? (parseFloat(cbSh.getRange('I2').getValue()) || 0) : 0;
    const result = {
      ok: true,
      cbOpen:     cbOpen,
      cb:         readSheet(SH_CB,       CB_FIELDS),
      iou:        readSheet(SH_IOU,      IOU_FIELDS),
      vouchers:   readSheet(SH_VOUCHERS, V_FIELDS),
      cashCounts: readCashCount(),
      meta:       readMeta()
    };
    return respond(result);
  } catch(err) {
    return respond({ok: false, error: err.message});
  }
}

function respond(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}

// ── Generic sheet write (data starts at row 2) ───────
function writeSheet(name, rows, fields) {
  const sh = SS.getSheetByName(name);
  if (!sh) throw new Error('Sheet not found: ' + name);
  const last = sh.getLastRow();
  if (last > 1) sh.getRange(2, 1, last - 1, fields.length).clearContent();
  if (!rows || !rows.length) return;
  sh.getRange(2, 1, rows.length, fields.length)
    .setValues(rows.map(r => fields.map(f => r[f] !== undefined ? r[f] : '')));
}

// ── Generic sheet read ────────────────────────────────
function readSheet(name, fields) {
  const sh = SS.getSheetByName(name);
  if (!sh) throw new Error('Sheet not found: ' + name);
  const last = sh.getLastRow();
  if (last < 2) return [];
  return sh.getRange(2, 1, last - 1, fields.length).getValues()
    .filter(r => r[0] !== '' && r[0] !== null)
    .map(r => {
      const o = {id: Date.now() + Math.random()}; // assign local id on load
      fields.forEach((f, i) => o[f] = r[i]);
      if (o.amount  !== undefined) o.amount  = parseFloat(o.amount)  || 0;
      if (o.rupees  !== undefined) o.rupees  = parseFloat(o.rupees)  || 0;
      if (o.cents   !== undefined) o.cents   = parseInt(o.cents)     || 0;
      return o;
    });
}

// ── CashCount sheet: denominations in col A, counts in col B ──
function writeCashCount(counts) {
  const sh = SS.getSheetByName(SH_CC);
  if (!sh) throw new Error('Sheet not found: ' + SH_CC);
  // Write denomination labels to col A (A2:A12) and counts to col B (B2:B12)
  const vals = CC_DENOMS.map(d => [d, counts[d] !== undefined ? counts[d] : 0]);
  sh.getRange(2, 1, vals.length, 2).setValues(vals);
}

function readCashCount() {
  const sh = SS.getSheetByName(SH_CC);
  if (!sh || sh.getLastRow() < 2) return {};
  const data = sh.getRange(2, 1, CC_DENOMS.length, 2).getValues();
  const out = {};
  data.forEach(r => { if (r[0]) out[r[0]] = parseInt(r[1]) || 0; });
  return out;
}

// ── Meta sheet ────────────────────────────────────────
function writeMeta(m) {
  if (!m) return;
  const sh = SS.getSheetByName(SH_META);
  if (!sh) throw new Error('Sheet not found: ' + SH_META);
  sh.getRange('A1:B1').setValues([['key', 'value']]);
  sh.getRange('A2:B4').setValues([
    ['cbOpen', m.cbOpen  || 0],
    ['lastCC', JSON.stringify(m.lastCC || null)],
    ['ccHist', JSON.stringify(m.ccHist || [])]
  ]);
}

function readMeta() {
  const sh = SS.getSheetByName(SH_META);
  if (!sh || sh.getLastRow() < 2) return { cbOpen: 0, lastCC: null, ccHist: [] };
  const data = sh.getRange('A2:B4').getValues();
  const m = {};
  data.forEach(r => { if (r[0]) m[r[0]] = r[1]; });
  return {
    cbOpen: parseFloat(m.cbOpen) || 0,
    lastCC: m.lastCC ? JSON.parse(m.lastCC) : null,
    ccHist: m.ccHist ? JSON.parse(m.ccHist) : []
  };
}
